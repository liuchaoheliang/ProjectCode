create table trans
(
    id int primary key not null AUTO_INCREMENT,

	bill_no varchar(32),##账单平台号
	trans_sn varchar(32),##交易编号

	currency_value_real_all varchar(32),##总实际货币数值，由规则计算出,买家实际支付金额
	
	bank_points_account varchar(32),##银行积分账户

	bank_points_value_real_all varchar(32),##银行总实际积分数值，由规则计算出
	fft_points_account varchar(32),##分分通积分账户
	fft_points_value_all varchar(32),##分分通总积分值，所有交易明细累计
	fft_points_value_real_all varchar(32),##分分通总实际积分数值，由规则计算出

	failure_time varchar(32),##失效时间
	state varchar(32),##状态
	create_time varchar(32),##创建时间
	update_time varchar(32),##更新时间
	remark varchar(100)##备注
) AUTO_INCREMENT=1001000000000001;
